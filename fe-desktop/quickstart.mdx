## Start the development server

After [installation](/fe-desktop/installation), start the local development server:

```bash
pnpm start
```

The application will be available at [http://localhost:8080](http://localhost:8080).

<Info>
  The dev server includes hot module replacement (HMR) for instant updates during development.
</Info>

## Project structure overview

```
fe-desktop/
├── src/
│   ├── app/              # Core application setup
│   ├── components/       # Reusable UI components
│   ├── modules/          # Feature modules
│   │   ├── Betslip/      # Betting slip functionality
│   │   ├── LayoutMain/   # Main layout structure
│   │   └── LobbyNavigation/ # Navigation sidebar
│   ├── pages/            # Route-based pages
│   ├── lib/              # Core libraries
│   │   ├── api/          # API clients
│   │   └── mobx/         # State management
│   └── theme/            # Styling and themes
```

## Basic workflow

### 1. Run the application

```bash
pnpm start
```

### 2. Make changes

Edit any file in the `src/` directory. Changes will hot-reload automatically.

### 3. Build for production

```bash
pnpm build
```

The production build will be created in the `dist/` folder.

## Example: Adding a new component

Let's create a simple component to display event information:

<Steps>
  <Step title="Create the component file">
    Create `src/components/EventInfo/EventInfo.tsx`:
    
    ```typescript
    import { observer } from 'mobx-react-lite';
    import './EventInfo.scss';
    
    interface EventInfoProps {
      eventName: string;
      startTime: string;
      competitors: string[];
    }
    
    export const EventInfo = observer(({ eventName, startTime, competitors }: EventInfoProps) => {
      return (
        <div className="event-info">
          <h3 className="event-info__title">{eventName}</h3>
          <time className="event-info__time">{startTime}</time>
          <div className="event-info__competitors">
            {competitors.map((competitor, index) => (
              <span key={index} className="event-info__competitor">
                {competitor}
              </span>
            ))}
          </div>
        </div>
      );
    });
    ```
  </Step>
  
  <Step title="Add styles">
    Create `src/components/EventInfo/EventInfo.scss`:
    
    ```scss
    .event-info {
      padding: 1rem;
      border: 1px solid var(--border-color);
      border-radius: 8px;
      
      &__title {
        font-size: 1.25rem;
        font-weight: 600;
        margin-bottom: 0.5rem;
      }
      
      &__time {
        display: block;
        color: var(--text-secondary);
        margin-bottom: 1rem;
      }
      
      &__competitors {
        display: flex;
        gap: 0.5rem;
      }
      
      &__competitor {
        padding: 0.25rem 0.75rem;
        background: var(--bg-secondary);
        border-radius: 4px;
      }
    }
    ```
  </Step>
  
  <Step title="Create index file">
    Create `src/components/EventInfo/index.ts`:
    
    ```typescript
    export { EventInfo } from './EventInfo';
    ```
  </Step>
  
  <Step title="Use the component">
    Import and use in any page or component:
    
    ```typescript
    import { EventInfo } from '@/components/EventInfo';
    
    function MyPage() {
      return (
        <EventInfo
          eventName="Champions League Final"
          startTime="2024-05-25 20:00"
          competitors={["Team A", "Team B"]}
        />
      );
    }
    ```
  </Step>
</Steps>

## Working with MobX stores

Access global state using MobX stores:

```typescript
import { observer } from 'mobx-react-lite';
import { useStore } from '@/lib/mobx';

export const BetslipCounter = observer(() => {
  const { betslip } = useStore();
  
  return (
    <div>
      <p>Bets in slip: {betslip.bets.length}</p>
      <p>Total odds: {betslip.totalOdds.toFixed(2)}</p>
    </div>
  );
});
```

## Making API calls

Use the HTTP client for API requests:

```typescript
import { httpClient } from '@/lib/api';

async function fetchEvents() {
  try {
    const response = await httpClient.get('/events', {
      params: {
        discipline: 'football',
        live: true
      }
    });
    return response.data;
  } catch (error) {
    console.error('Failed to fetch events:', error);
  }
}
```

## WebSocket subscriptions

Subscribe to real-time updates:

```typescript
import { useSocketEntitySubscription } from '@/helpers/useSocketEntitySubscription';

function LiveEventComponent({ eventId }) {
  useSocketEntitySubscription('event', eventId);
  
  // Component will automatically receive updates
  return <div>Event {eventId}</div>;
}
```
